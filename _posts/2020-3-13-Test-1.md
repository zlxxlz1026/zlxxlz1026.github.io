#include<bits/stdc++.h>
using namespace std;
class Solution {
public:
    void moveZeroes(vector<int>& nums) {
    	int tmp = 0;
    	for(int i=0;i<nums.size();i++){
    		if(nums[i]!=0){
    			nums[tmp] = nums[i];
    			tmp++;
    		}
    	}
    	for(int i=tmp;i<nums.size();i++)
    		nums[i] = 0;
    }
};
